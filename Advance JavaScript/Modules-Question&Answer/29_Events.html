29. How to used JavaScript Events to do all examples?

Code:


               
               <button id="clickButton">Click me</button>

                
                <div id="hoverDiv">Hover over me</div>

                
                <input type="text" id="keyInput" placeholder="Press any key" />

                
                <form id="submitForm">
                    <input type="text" />
                    <input type="submit" />
                </form>

                
                <img id="loadImage" src=""/>



                <script>
                    // Click Event
                    document.getElementById("clickButton").onclick = function () {
                      alert("Button clicked!");
                    };
              
                    // Mouseover Event
                    document.getElementById("hoverDiv").onmouseover = function () {
                      alert("Mouse over!");
                    };
              
                    // Keydown Event
                    document.getElementById("keyInput").onkeydown = function (event) {
                      alert("Key pressed: " + event.key);
                    };
              
                    // Submit Event
                    document.getElementById("submitForm").onsubmit = function (event) {
                      event.preventDefault(); // Prevent form submission
                      alert("Form submitted!");
                    };
              
                    // Load Event
                    document.getElementById("loadImage").onload = function () {
                      alert("Image loaded!");
                    };
              
                    // Resize Event
                    window.onresize = function () {
                      alert("Window resized!");
                    };
              
                    // Scroll Event
                    window.onscroll = function () {
                      alert("Scrolled!");
                    };
                </script>
