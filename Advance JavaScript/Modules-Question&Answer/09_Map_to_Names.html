Code:

            <script>
                let john = { name: "John", age: 25 };
                let pete = { name: "Pete", age: 30 };
                let mary = { name: "Mary", age: 28 };
                let users = [john, pete, mary];
        
                let names = users.map(user => user.name);
        
                alert(names.join(", ")); // John, Pete, Mary
            </script>
