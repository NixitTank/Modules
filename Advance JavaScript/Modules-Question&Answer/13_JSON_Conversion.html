13. Turn the object into JSON and back Turn the user into JSON and then read it back into another variable. user = { name: "John Smith", age: 35};

Code:

                var user = { name: "John Smith", age: 35 };
                
                var userJSON = JSON.stringify(user);
                console.log("JSON String:", userJSON); // Output: {"name":"John Smith","age":35}
        
                document.getElementById("Ans").innerHTML += "<p>JSON String: " + userJSON + </p>";
        
                var parsedUser = JSON.parse(userJSON);
                console.log("Parsed Object:", parsedUser); // Output: { name: 'John Smith', age: 35 }
        
                document.getElementById("Ans").innerHTML += "Parsed Object: Name - " + parsedUser.name + ", Age - " + parsedUser.age ;
