15. Create a program that will select all the classes and loop over and whenever i click the button the alert should show

Code:

                <button class="alert-button">Button 1</button>
                <button class="alert-button">Button 2</button>
                <button class="alert-button">Button 3</button>


                <script>

                    document.addEventListener('DOMContentLoaded', () => {
                        const buttons = document.querySelectorAll('.alert-button');
        
                        buttons.forEach(button => {
                            button.addEventListener('click', () => {
                                alert('Button clicked!');
                            });
                        });
                    });
        
                </script>
