36. Write a program you have to make a summation of first and last Digit. (E.g., 1234 Ans: - 5) in JS?

Program :


                <h3 id="ans1"></h3>
                <h3 id="ans2"></h3>
                <h3 id="ans3"></h3>
                <h3 id="ans4"></h3>

                <script>

                    let msg = alert('Sum of First Digit Last Digit');
                    var num = prompt('Enter Digit Minimum Three : ');
                    var LastDigit = num % 10 ;
                    var len = num.toString().length-1;
        
                    FirstDigit = parseInt(num/Math.pow(10,len));
                    var sum = FirstDigit+LastDigit;
        
                    document.getElementById('ans1').innerHTML = ("Your Entered  Number : " + num + "<br>");
                    document.getElementById('ans2').innerHTML = ("FirstDigit : " + FirstDigit + "<br>");
                    document.getElementById('ans3').innerHTML = ("LastDigit : " + LastDigit  + "<br>");
                    document.getElementById('ans4').innerHTML = ("Sum of Number : " + sum);
                    
                </script>
