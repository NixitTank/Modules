25 Write to find number is even or odd using ternary operator in JS? 

            <h3 id="ans1"></h3>
            <h3 id="ans2"></h3>

            <script>
                const number = prompt("Enter a number: ");
        
                if(number % 2 == 0) {
                    document.getElementById('ans1').innerHTML = (number + " is Even Number.");
                }
        
                else {
                    document.getElementById('ans2').innerHTML = (number + " is Odd Number.");
                }
            </script>
