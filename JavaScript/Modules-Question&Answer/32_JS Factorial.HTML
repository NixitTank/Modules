32. Write to print factorial of given number?

Program :


            <h3 id="ans"></h3>

            <script>
                let num = prompt('Enter the number for Factorial Calculation = ')
                function factorial(n) { 
                    if (n == 0) return 1; 
                    return n * factorial(n - 1); 
                  } 
                    
                  // Driver Code 
                  //let num = 5; 
                  document.getElementById('ans').innerHTML = ("Factorial of " + num + " is " + factorial(num)); 
            </script>
