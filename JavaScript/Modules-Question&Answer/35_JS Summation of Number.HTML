35. Write a program make a summation of given number (E.g., 1523 Ans: - 11) in JS?

Program :


            <h3 id="ans"></h3>

            <script>

                var n = prompt('Enter Number : ');
                function getSum(n)
                {
                    var sum = 0;
                    while (n != 0) {
                        sum = sum + n % 10;
                        n = parseInt(n / 10);
                    }
                    return sum;
                }
                document.getElementById('ans').innerHTML = (getSum(n));
                
            </script>
