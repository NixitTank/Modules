34. Write to print number in reverse order e.g.: number = 64728 --> reverse =82746 in JS?

Program :


            <h3 id="ans"></h3>

            <script>
                let num = prompt('Enter Number :' );
                function reverseFunction(num) {
                    let numStr = num.toString();
                    let reversedStr = '';
                    for (let i = numStr.length - 1; i >= 0; i--) {
                        reversedStr += numStr[i];
                    }
                    return parseInt(reversedStr);
                }
                
                let reversedNum = reverseFunction(num);
                
                document.getElementById('ans').innerHTML = (reversedNum);
                
          </script>
