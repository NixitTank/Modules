33. Write to print Fibonacci series up to given numbers?

Program :


            <h3 id="ans"></h3>

            <script>

                num = prompt('Enter Number = ')
        
                function fibonacci(num) {
                  let num1 = 0;
                  let num2 = 1;
                  let sum;
                  let i = 0;
                  for (i = 0; i < num; i++) {
                      sum = num1 + num2;
                      num1 = num2;
                      num2 = sum;
                  }
                  return num2;
                }
        
                document.getElementById('ans').innerHTML = fibonacci(num);
                    
            </script>
