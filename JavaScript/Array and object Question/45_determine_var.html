45. Write a JavaScript program to determine if a variable is array?


            function isArray(variable) {
                return Array.isArray(variable);
              }
              
                  let testArray = [1, 2, 3];
              let testVariable = "hello";
              
              console.log("Is testArray an array? " + isArray(testArray)); // Output: true
              console.log("Is testVariable an array? " + isArray(testVariable)); // Output: false
